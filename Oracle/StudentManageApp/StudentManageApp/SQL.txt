과정별 학생관리 프로젝트

학생번호 이름 나이 과정명 대학교 전공 이메일 전화번호 특징 상담


STUDENT_BY_COURSE (
	학생번호	NUMBER(5)	PRIMARY KEY,
	이름	VARCHAR2(20)	NOT NULL,
	나이	NUMBER(3),
	과정명	VARCHAR2(30),
	대학교	VARCHAR2(20),
	전공	VARCHAR2(20),
	이메일	VARCHAR2(30),
	전화번호	VARCHAR2(13),
	특징	VARCHAR2(100),
	상담	VARCHAR2(200)	
);

CREATE SEQUENCE  STUDENT_SEQ2 
	MINVALUE 1 
	MAXVALUE 99999 
	INCREMENT BY 1 
	START WITH 1 
	CACHE 2 
	NOORDER  
	NOCYCLE ;
